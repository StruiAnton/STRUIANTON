void vodrandom(int N) {
	ofstream f;
	f.open(F1, ios::out);
	for (int i = 0; i < N; i++) {
		int j = rand() % N + 1;
		f << j << " ";
	}
	f.close();
}

void vodpovozr(int N) {
	ofstream f;
	f.open(F1, ios::out);
	for (int i = 0; i < N; i++) {
		f << i + 1 << " ";
	}
	f.close();
}

void vodpoybB(int N) {
	ofstream f;
	f.open(F1, ios::out);
	for (int i = 0; i < N; i++) {
		int j = N - i;
		f << j << " ";
	}
	f.close();
}


int fibonachi(int n) {
	if (n == 0) {
		return 0;
	}
	if (n == 1 || n == 2) {
		return 1;
	}
	return fibonachi(n - 1) + fibonachi(n - 2);
}
