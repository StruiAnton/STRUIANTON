class amuni {
protected:
	string name;
	int money;
	int mas;
public:
	virtual void show() = 0;
	virtual int get_money() = 0;
	virtual int get_mas() = 0;
	virtual void print() = 0;
};

void cinn(amuni *A[5]);
void sum(amuni *A[5]);
void print(amuni *A[5]);
void poisk(amuni *A[5]);

template <typename V>
void sort(V *A[5]) {
	for (int i = 0; i < 5; i++)
		for (int j = 0; j < 4; j++) {
			if (A[j]->get_mas() > A[j + 1]->get_mas()) {
				swap(A[j], A[j + 1]);
			}
		}
}

void cinn(amuni *A[5]) {

	A[4] = new cap;
	A[4]->show();

void sum(amuni *A[5]) {
	int sum = 0;
	for (int i = 0; i < 5; i++) {
		sum += A[i]->get_money();
	}
	cout << "Стоимость всех вещей равна " << sum << endl;
}

void print(amuni *A[5]) {
	for (int i = 0; i < 5; i++) {
		A[i]->print();
	}
}

void poisk(amuni *A[5]) {
	cout << "Введите начальную цену диапазона" << endl;
	int k;
	cin >> k;
	cout << "Введите конечную цену диапазона" << endl;
	int l;
	cin >> l;
	for (int i = 0; i < 5; i++) {
		if (A[i]->get_money() >= k && A[i]->get_money() <= l) {
			A[i]->print();
		}
	}
}
