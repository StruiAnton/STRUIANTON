#include "pch.h"
#include <iostream>
#include <math.h>

using namespace std;

int a(int N)
{
	int d= 1;
	while (++d != N)
	{
		if (N % d == 0)    return 1;
	}
	return 0;
}

int main() {
	setlocale(0, "");
	int N;
	cin >> N;
	if (a(N) == 0) {
		cout << "Простое" << endl;
	}
	else {
		cout << "Cоставное" << endl;
	}
	bool pr = true;
	for (int i = 2; i <= sqrt(N); i++)
	{
		if (N%i == 0)
			pr = false;
	}
	if (pr == true)
		cout << "Простое";
	else
		cout << "Составное";
	return 0;
}
